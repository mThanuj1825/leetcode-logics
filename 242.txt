class Solution:
    def isAnagram(self, s: str, t: str) -> bool:
        m = {}

        for c in s:
            m[c] = m.get(c, 0) + 1
        
        for c in t:
            if c not in m:
                return False
            m[c] -= 1
        
        for k in m:
            if m[k] != 0:
                return False

        return True
